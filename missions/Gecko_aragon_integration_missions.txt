aragon_integration_slot_4 = {
	slot = 4
	generic = no
	ai = yes
	potential = {
		AND = {
			tag = SPA
			NOT = { primary_culture = aragonese }
			NOT = { primary_culture = catalan }
		}
	}
	has_country_shield = yes

	gc_ara_consultate_of_the_sea = {
		position = 7
		icon = mission_consulate_of_the_sea
		provinces_to_highlight = {
			OR = {
				province_id = 109
				province_id = 1295
				AND = {
					OR = {
						province_id = 213
						province_id = 220
					}
					OR = {
						NOT = { X_has_latest_building_trigger = { builder = ROOT trade = yes } }
						NOT = { owned_by = ROOT }
					}
				}
			}
		}
		trigger = {
			1295 = {
				trade_share = {
					country = ROOT
					share = 70
				}
			}
			109 = {
				trade_share = {
					country = ROOT
					share = 15
				}
			}
			213 = {
				X_has_latest_building_trigger = { builder = ROOT trade = yes }
				owned_by = ROOT
			}
			220 = {
				X_has_latest_building_trigger = { builder = ROOT trade = yes }
				owned_by = ROOT
			}
		}
		effect = {
			set_country_flag = ara_consulate_flag
			country_event = { id = flavor_ara.1 }
			liguria_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			corsica_sardinia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			corsica_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	gc_ara_valencian_silk = {
		position = 8
		icon = mission_valencian_silk
		required_missions = { gc_ara_consultate_of_the_sea }
		provinces_to_highlight = {
			province_id = 220
		}
		trigger = {
			owns_core_province = 220
			if = {
				limit = {
					has_dlc = "Dharma"
				}
				220 = {
					OR = {
						province_has_center_of_trade_of_level = 3
						development = 20
					}
				}
			}
			else = {
				220 = {
					development = 20
				}
			}
		}
		effect = {
			country_event = { id = X_SPA_events.1 }
		}
	}
	gc_ara_industry = {
		position = 9
		icon = mission_industria_sedera
		required_missions = { gc_ara_valencian_silk  }
		trigger = {
			num_of_owned_provinces_with = {
				value = 10
				X_has_latest_building_trigger = { builder = ROOT production = yes }
				OR = {
					trade_goods = wine
					trade_goods = glass
					trade_goods = paper
					trade_goods = silk
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "industria_sedera"
				duration = 7300
			}
		}
	}
	gc_ara_golden_age = {
		position = 11
		icon = mission_golden_century
		required_missions = { gc_ara_industry gc_ara_banking }
		trigger = {
			prestige = 80
			share_of_starting_income = 4
		}
		effect = {
			add_country_modifier = {
				name = "ara_golden_age"
				duration = 36500
			}
		}
	}
}

	}

	gc_ara_trade_guilds = {
		position = 9
		icon = mission_guilds_of_barcelona
		required_missions = { gc_ara_valencian_silk }
		provinces_to_highlight = {
			province_id = 213
		}
		trigger = {
			owns_core_province = 213
			213 = {
				development = 20
				X_has_latest_building_trigger = { builder = ROOT navy_forcelimit = yes }
			}
			if = {
				limit = {
					has_estate = estate_burghers
				}

					estate = estate_burghers

				}
				estate_influence = {
					estate = estate_burghers
					influence = 40
				}
				num_of_estate_privileges = {
					estate = estate_burghers
					value = 2
				}
			}
			else = {
				mercantilism = 35
			}
		}
		effect = {
			213 = {
				add_base_production = 4
				add_province_modifier = {
					name = "barcelona_guilds"
					duration = -1
				}
				add_province_modifier = {
					name = X_SPA_royal_shipyards_of_barcelona_pmod

				}
			}
		}
	}
	gc_ara_banking = {
		position = 10
		icon = mission_taula_de_canvi

		trigger = {
			if = {
				limit = {
					has_estate = estate_burghers
				}
				estate_loyalty = {
					estate = estate_burghers
					loyalty = 75
				}
				estate_influence = {
					estate = estate_burghers
					influence = 60
				}
				num_of_estate_privileges = {
					estate = estate_burghers
					value = 4
				}
			}
			else = {
				mercantilism = 50
			}
			treasury = 3000
		}
		effect = {
			add_country_modifier = {
				name = "taula_de_canvi"
				duration = 7300
			}
		}
	}
}
