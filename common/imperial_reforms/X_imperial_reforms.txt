X_CRI_reform_3 = {
	empire = celestial_empire
	on_effect = {
		custom_tooltip = X_CRI_reform_3_tt
		hidden_effect = {
			every_province = {
				limit = { 
					has_terrain = steppe
					owner = { is_emperor_of_china = yes }
				}
				add_province_modifier = {
					name = X_CRI_reform_3_pmod
					duration = -1
				}
			}
		}
		set_global_flag = X_CRI_reform_3_gflag
	}
	off_effect = {
		custom_tooltip = X_CRI_reform_3_tt2
		hidden_effect = {
			every_province = {
				limit = { has_province_modifier = X_CRI_reform_3_pmod }
				remove_province_modifier = X_CRI_reform_3_pmod
			}
		}
		clr_global_flag = X_CRI_reform_3_gflag
	}
}

X_CRI_reform_8 = {
	empire = celestial_empire
	emperor = {
		years_of_nationalism = -5
	}
	on_effect = {
		custom_tooltip = X_CRI_reform_8_tt
		set_global_flag = X_CRI_reform_8_gflag
		hidden_effect = {
			every_country = {
				limit = { is_emperor_of_china = yes }
				change_tag = GLH
				if = {
					limit = { NOT = { government_rank = 3 } }
					set_government_rank = 3
				}
				country_event = { id = ideagroups.1 }
				every_province = {
					limit = {
						NOT = { country_or_non_sovereign_subject_holds = PREV }
						culture_group = tartar
					}
					add_permanent_claim = PREV
				}
			}
		}
	}
	off_effect = {
		custom_tooltip = X_CRI_reform_9_tt2
		clr_global_flag = X_CRI_reform_9_gflag
	}
}
