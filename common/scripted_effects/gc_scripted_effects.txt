change_government_to_gc_ethiopia_Government = {
	if = {
		limit = {
			NOT = { government = gc_ethiopian_government }
		}
		if = {
			limit = {
				government = republic
			}
			change_government = gc_ethiopian_government
			lose_reforms = 4
		}
		else = {
			change_government = gc_ethiopian_government
			lose_reforms = 2
		}
	}
}
