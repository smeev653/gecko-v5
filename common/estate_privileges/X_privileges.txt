X_estate_cossacks_privilege_1 = {
    icon = privilege_demand_more_troops
	#land_share = 5
    loyalty = 0.05            
    influence  = 0.1      
	is_valid = {
		any_owned_province = {
			OR = {
				has_terrain = steppe
				AND = {
					region = ruthenia_region
					OR = {
						has_terrain = farmlands
						has_terrain = grasslands
					}
				}
			}
		}
	}
	on_granted = {
		custom_tooltip = X_estate_cossacks_privilege_1_on_granted_tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_terrain = steppe
						AND = {
							region = ruthenia_region
							OR = {
								has_terrain = farmlands
								has_terrain = grasslands
							}
						}
					}
					has_fort_building_trigger = yes
				}
				add_province_modifier = {
					name = X_estate_cossacks_privilege_1_pmod
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = X_estate_cossacks_privilege_1_on_revoked_tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = X_estate_cossacks_privilege_1_pmod
				}
				remove_province_modifier = X_estate_cossacks_privilege_1_pmod
			}
		}
	}
	on_invalid = {
		custom_tooltip = X_estate_cossacks_privilege_1_on_revoked_tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = X_estate_cossacks_privilege_1_pmod
				}
				remove_province_modifier = X_estate_cossacks_privilege_1_pmod
			}
		}
	}	
    benefits = {
		reinforce_speed = 0.20
	}
	ai_will_do = {
		factor = 0
	}
	cooldown_years = 20
	on_cooldown_expires = {
		remove_estate_privilege = X_estate_cossacks_privilege_1
	}
}
