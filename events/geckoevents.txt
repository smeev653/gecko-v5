country_event = { #Poland Ruler Failsafe
	id = gecko.1
	title = gecko.1.t
	desc = gecko.1.d
	picture = CIVIL_WAR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = POL
		NOT = { adm = 5 dip = 4 mil = 6 }
		NOT = { is_year = 1475 }
		is_free_or_tributary_trigger = yes
		has_country_flag = gecko_localnoble
	}

	 immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
					is_in_capital_area = yes
				}
				save_event_target_as = gecko_pol_pretenders
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = gecko_pol_nobles
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = gecko_pol_nobles2
			}
		}
	 }

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = gecko.1.a
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
		define_ruler = {
			culture = ROOT
			religion = ROOT
			adm = 5
			dip = 4
			mil = 6
			age = 34
			claim = 20
		}
		event_target:gecko_pol_nobles = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		event_target:gecko_pol_nobles2 = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
	}

	option = {
		name = gecko.1.b
		add_legitimacy = 20
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		event_target:gecko_pol_pretenders = {
			spawn_rebels = {
				type = pretender_rebels
				size = 2
			}
		}
	}
}

	desc = gecko.8.d

			OR = {
				 has_estate_privilege = estate_gc_sipahi 
				 has_estate_privilege = estate_gc_noble_levies
			} 

	}
	picture = MAMLUK_WARRIORS_eventPicture

	is_triggered_only = yes

	option = {
		name = gecko.8.a
	}
	option = {
		name = gecko.8.b
		if = {
			limit = {
				has_estate = estate_nomadic_tribes
			}
			add_estate_loyalty = {
				estate = estate_nomadic_tribes
				loyalty = -30
			}
			capital_scope = { 
				spawn_rebels = {
						type = tribal_rebels
						size = 1
					}
			}
			remove_estate_privilege = estate_gc_noble_levies
			remove_estate_privilege = estate_gc_sipahi
		}
		if = {
			limit = {
				has_estate = estate_nobles
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -30
			}
			capital_scope = { 
				spawn_rebels = {
						type = noble_rebels
						size = 1
					}
			}
			remove_estate_privilege = estate_gc_noble_levies
			remove_estate_privilege = estate_gc_sipahi
		}		
	}
}
