namespace = flavor_pol

#Election of Kasimir
country_event = {
	id = flavor_pol.18
	title = flavor_pol.18.t
	desc = flavor_pol.18.d
	picture = LIBERUM_VETO_eventPicture
	fire_only_once = yes
	major = yes
	trigger = {
		tag = POL
		is_subject = no
		has_regency = yes
		LIT = {
			exists = yes
			is_subject = no
			NOT = { junior_union_with = POL }
			dynasty = "Jagiellon"
		}
		NOT = { is_year = 1450 }
	}
	immediate = {
		set_global_flag = plc_union_outcome_flag
	}
	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 0.5
			is_year = 1448
		}
	}
	option = {
		name = flavor_pol.18.a
		define_ruler = {
			change_adm = 3
			change_dip = 3
			change_mil = 4
			claim = 100
			dynasty = "Jagiellon"
			culture = lithuanian
		}
		set_country_flag = gecko_localnoble
		create_union = LIT
		LIT = {
			release = PLT
			release = SMO
			release = VOL
			hidden_effect = {
				every_core_province = {
					limit = {
						OR = {
							is_core = PLT
							is_core = SMO
							is_core = VOL
						}
					}
					remove_core = LIT
				}
			}
		}
		inherit = VOL
	}
	after = { country_event = { id = flavor_pol.20 } }
}

# Sindicat Remença
country_event = {
	id = flavor_ara.6
	title = "flavor_ara.EVTNAME6"
	desc = "flavor_ara.EVTDESC6"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SPA
		OR = { 
			adm = 3
			dip = 3
		}
		any_owned_province = {
			culture = catalan
		}
		NOT = { 
			any_owned_province = {
				has_province_modifier = ara_sindicat_remenca 
			}
		}
	}
	
	mean_time_to_happen = {
		months = 100
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
				}
				set_province_flag = ara_sindicat_remença_flag
			}
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
					NOT = {  has_province_flag = ara_sindicat_remença_flag }
				}
				set_province_flag = ara_sindicat_remença_flag
			}
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
					NOT = { has_province_flag = ara_sindicat_remença_flag }
				}
				set_province_flag = ara_sindicat_remença_flag
			}
		}
	}
	
	option = {
		name = "flavor_ara.EVTOPTA6"
		every_owned_province = {
			limit = {
				has_province_flag = ara_sindicat_remença_flag
			}
			add_local_autonomy = 25
			add_province_modifier = { 
				name = ara_sindicat_remenca 
				duration = -1
			}
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_flag = ara_sindicat_remença_flag
				}
				clr_province_flag = ara_sindicat_remença_flag
			}
		}
	}
	option = {
		name = "flavor_ara.EVTOPTB6"
		every_owned_province = {
			limit = {
				has_province_flag = ara_sindicat_remença_flag
			}
			clr_province_flag = ara_sindicat_remença_flag
			spawn_rebels = {
				type = anti_tax_rebels
				size = 1
			}
		}
	}
}
